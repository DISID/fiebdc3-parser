19  package com.disid.fiebdc3.antlr4;
20  
21  import org.antlr.v4.runtime.DefaultErrorStrategy;
22  import org.antlr.v4.runtime.Parser;
23  import org.antlr.v4.runtime.RecognitionException;
24  import org.antlr.v4.runtime.Token;
25  
26  /**
27   * Detect if there has been any parse errors, so the tests fail.
28   * 
29   * @author DiSiD Team
30   */
31  public class FailTestOnErrorStrategy extends DefaultErrorStrategy {
32      
33      private boolean hasErrors = false;
34  
35      @Override
36      public void reportError(Parser recognizer, RecognitionException e) {
37  	hasErrors = true;
38          super.reportError(recognizer, e);
39      }
40      
41      @Override
42      public void recover(Parser recognizer, RecognitionException e) {
43  	hasErrors = true;
44          super.recover(recognizer, e);
45      }
46      
47      @Override
48      public Token recoverInline(Parser recognizer) throws RecognitionException {
49  	hasErrors = true;
50          return super.recoverInline(recognizer);
51      }
52      
53      public boolean isHasErrors() {
54  	return hasErrors;
55      }
56  }
