19  package com.disid.fiebdc3.antlr4;
20  
21  import static org.junit.Assert.assertFalse;
22  
23  import java.io.File;
24  import java.io.FileReader;
25  import java.net.URL;
26  
27  import org.antlr.v4.runtime.ANTLRInputStream;
28  import org.antlr.v4.runtime.CommonTokenStream;
29  import org.antlr.v4.runtime.LexerNoViableAltException;
30  import org.antlr.v4.runtime.tree.ParseTree;
31  import org.junit.Before;
32  import org.junit.Test;
33  
34  /**
35   * Integration tests for the FIEBDC 3 file format ANTLR4 based parser.
36   * 
37   * @author DiSiD Team
38   */
39  public class Fiebdc3ParserIT {
40  
41      private Fiebdc3Parser parser;
42      private FailTestOnErrorStrategy errorStrategy;
43  
44      /**
45       * @throws java.lang.Exception
46       */
47      @Before
48      public void setUp() throws Exception {
49          URL testResource = this.getClass().getClassLoader()
50                  .getResource("test.bc3");
51          File testFile = new File(testResource.getPath());
52          // create a CharStream that reads from standard input
53          ANTLRInputStream input = new ANTLRInputStream(new FileReader(testFile));
54          // create a lexer that feeds off of input CharStream
55          Fiebdc3Lexer lexer = new Fiebdc3Lexer(input) {
56              @Override
57              public void recover(LexerNoViableAltException e) {
58                  throw new RuntimeException(e); // Bail out
59              }
60          };
61          // create a buffer of tokens pulled from the lexer
62          CommonTokenStream tokens = new CommonTokenStream(lexer);
63          parser = new Fiebdc3Parser(tokens);
64          errorStrategy = new FailTestOnErrorStrategy();
65          parser.setErrorHandler(errorStrategy);
66      }
67  
68      @Test
69      public void testParser() {
70          ParseTree tree = parser.database(); // begin parsing at database rule
71          System.out.println(tree.toStringTree(parser)); // print LISP-style tree
72          assertFalse("Found parse errors", errorStrategy.isHasErrors());
73      }
74  }
