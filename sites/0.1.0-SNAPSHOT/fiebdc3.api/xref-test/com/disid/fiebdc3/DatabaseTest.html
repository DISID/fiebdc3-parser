18  package com.disid.fiebdc3;
19  
20  import static org.junit.Assert.*;
21  
22  import org.junit.Before;
23  import org.junit.Test;
24  
25  /**
26   * Unit tests for the {@link Database} class.
27   * @author DiSiD Team
28   */
29  public class DatabaseTest {
30  
31      private Database database;
32  
33      @Before
34      public void setUp() throws Exception {
35         database = new Database();
36      }
37  
38      @Test
39      public void testGetConcept() {
40          Concept testConcept = database.getConcept("test");
41          assertNull(testConcept);
42  
43          testConcept = database.getOrAddConcept("test");
44  
45          assertNotNull(testConcept);
46          assertEquals("test", testConcept.getCode());
47  
48          testConcept = database.getConcept("test");
49          assertNotNull(testConcept);
50  
51          ConceptBreakdown bd = database.addConceptBreakdown("test");
52          assertEquals(testConcept, database.getConcept(bd));
53      }
54  
55      @Test
56      public void testAddConceptBreakdown() {
57          // Add an orphan ConceptBreakdown
58          ConceptBreakdown bd = database.addConceptBreakdown("test");
59          assertEquals(null, bd.getParentConceptCode());
60          assertEquals("test", bd.getConceptCode());
61          assertTrue(database.hasOrphanedConceptBreakdowns());
62  
63          // Add the orphan Breakdown to a parent one
64          ConceptBreakdown bd3 = database.addConceptBreakdown("parent", "test");
65          assertEquals("parent", bd3.getParentConceptCode());
66          assertEquals("test", bd3.getConceptCode());
67          assertTrue(database.hasOrphanedConceptBreakdowns());
68          database.setRoot("parent");
69          assertFalse(database.hasOrphanedConceptBreakdowns());
70  
71          // Add another child to the root ConceptBreakdown
72          ConceptBreakdown bd2 = database.addConceptBreakdown("parent", "child");
73          assertEquals("parent", bd2.getParentConceptCode());
74          assertEquals("child", bd2.getConceptCode());
75          assertFalse(database.hasOrphanedConceptBreakdowns());
76      }
77  
78      @Test
79      public void testAddMeasurement() {
80          Measurement measurement = database.createMeasurement();
81          assertTrue(database.hasOrphanedMeasurements());
82  
83          database.setMeasurement("test", measurement);
84          assertFalse(database.hasOrphanedMeasurements());
85  
86          ConceptBreakdown bd = database.addConceptBreakdown("test");
87          assertEquals(measurement, bd.getMeasurement());
88      }
89  
90  }
