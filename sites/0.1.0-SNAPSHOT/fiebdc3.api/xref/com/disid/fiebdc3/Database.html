27  /**
28   * Contains the data of a Fiebdc 3 file.
29   * <p>
30   * A database has is composed of a root {@link Concept}, which itself might be
31   * composed of child {@link Concept}s:
32   * </p>
33   * 
34   * <pre>
35   * database
36   * '-> root concept
37   *     |-> concept 1
38   *     |   '-> measurement
39   *     |       |-> line 1
40   *     |       |-> line 2
41   *     |       ...
42   *     |       '-> line n 
43   *     |-> concept 2
44   *     ...
45   *     '-> concept n
46   * </pre>
47   * 
48   * <p>
49   * This class is also a factory for the elements it includes, so {@link Concept}
50   * s and {@link Measurement}s can only be created through a {@link Database}.
51   * </p>
52   * 
53   * @author DiSiD Team
54   */
55  public class Database {
56  
57      /**
58       * Enumeration of the available charset options for the Fiebdc3 file.
59       * 
60       * @author DiSiD Team
61       */
62      public enum Charset {
63          ANSI("ANSI"), C850("850"), C437("437");
64          
65          private final String code;
66          
67          /**
68           * Create a new Charset with the given code
69           * 
70           * @param code
71           *            of the charset
72           */
73          private Charset(String code) {
74              this.code = code;
75          }
76  
77          @Override
78          public String toString() {
79              return code;
80          }
81  
82          /**
83           * Returns the Charset with the given code or null if not available.
84           * 
85           * @param code
86           *            of the charset
87           * @return the Charset with the code
88           */
89          public static Charset parseCharset(String code) {
90              for (Charset c : Charset.values()) {
91                  if (c.code.equals(code)) {
92                      return c;
93                  }
94              }
95              return null;
96          }
97      }

259     /**
260      * Returns the concept with the given code in the database.
261      * 
262      * @param code
263      *            of the {@link Concept}
264      * @return the concept with the given code
265      */
266     public Concept getConcept(String code) {
267         Concept concept = rootConcept == null ? null : rootConcept
268                 .getConcept(code);
269 
270         if (concept == null) {
271             concept = orphanConcepts.get(code);
272         }
273 
274         return concept;
275     }
276 
277     /**
278      * Add a new root concept for the database.
279      * 
280      * @param code
281      *            to identify the concept
282      * @return the new root concept
283      */
284     public Concept addRootConcept(String code) {
285         Concept concept = orphanConcepts.remove(code);
286         if (concept == null) {
287             concept = new Concept(code);
288         }
289         setRootConcept(concept);
290         return concept;
291     }
292 
293     /**
294      * Looks for an already available concept. If it does not exist, a new one
295      * is created in the orphan codes list.
296      * 
297      * @param code
298      *            of the concept
299      * @return the concept
300      */
301     public Concept getAddConcept(String code) {
302         Concept concept = getConcept(code);
303         if (concept == null) {
304             concept = addOrphanConcept(code);
305         }
306         return concept;
307     }
308 
309     /**
310      * Creates a new concept, which is included in a set of concepts still not
311      * organized: it is not a root concept, nor it has been added as a child
312      * concept.
313      * 
314      * @param code
315      *            of the concept
316      * @return the created concept
317      */
318     public Concept addOrphanConcept(String code) {
319         Concept concept = orphanConcepts.get(code);
320         if (concept == null) {
321             concept = new Concept(code);
322             orphanConcepts.put(code, concept);
323         }
324         return concept;
325     }
326 
327     /**
328      * Returns if there are any orphan concepts pending to be organized.
329      * 
330      * @return if there are any orphan concepts pending to be organized
331      */
332     public boolean hasOrphanedConcepts() {
333         return orphanConcepts != null && !orphanConcepts.isEmpty();
334     }
335 
336     /**
337      * Returns if there are any orphan measurements pending to be setted to a
338      * concept.
339      * 
340      * @return if there are any orphan measurements pending to be setted to a
341      *         concept
342      */
343     public boolean hasOrphanedMeasurements() {
344         return orphanMeasurements != null && !orphanMeasurements.isEmpty();
345     }
346 
347     /**
348      * Creates a new concept, which is added as a child of the given concept.
349      * 
350      * @param code
351      *            of the concept
352      * @return the created concept
353      */
354     public Concept addChildConcept(String code, Concept parentConcept) {
355         Concept concept = orphanConcepts.remove(code);
356         if (concept == null) {
357             concept = new Concept(code);
358         }
359         parentConcept.addChildConcept(concept);
360         return concept;
361     }
362 
363     /**
364      * Creates a new measurement and adds it to the database, without bein
365      * setted to a concept.
366      * 
367      * @return the new Measurement
368      */
369     public Measurement addMeasurement() {
370         Measurement measurement = new Measurement();
371         orphanMeasurements.add(measurement);
372         return measurement;
373     }
374 
375     /**
376      * Sets a Measurement to the related Concept.
377      * 
378      * @param conceptCode
379      *            code of the Concept to set the Measure to
380      * @param measurement
381      *            to set to the Concept
382      */
383     public void setMeasurement(String conceptCode, Measurement measurement) {
384         Concept concept = getAddConcept(conceptCode);
385         concept.setMeasurement(measurement);
386         orphanMeasurements.remove(measurement);
387     }

389     @Override
390     public String toString() {
391         return "Fiebdc3Database {" + "File property of: " + fileProperty
392                 + ", Format version: " + fileFormat + ", Generated by: "
393                 + generatedBy + ", Header: " + header + ", Charset: " + charset
394                 + ", Comments: " + comments + ", Information type: " + infoType
395                 + ", Certification number: " + certNum
396                 + ", Certification date: " + certDate + ", Root concept: \n\t"
397                 + rootConcept + "}";
398     }
