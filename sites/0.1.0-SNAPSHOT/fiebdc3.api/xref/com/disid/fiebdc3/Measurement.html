19  package com.disid.fiebdc3;
20  
21  import java.util.ArrayList;
22  import java.util.List;
23  
24  /**
25   * Data of a measurement of a Fiebdc3 database.
26   * 
27   * @author DiSiD Team
28   */
29  public class Measurement {
30  
31      String parentConcept;
32  
33      String concept;
34  
35      List<Integer> position = new ArrayList<Integer>();
36  
37      Float total;
38  
39      List<Line> lines = new ArrayList<Line>();
40  
41      String label;
42  
43      Measurement() {
44          // Nothing to do
45      }
46  
47      /**
48       * Spec definition:<br/>
49       * <i>CODIGO_PADRE: CODIGO del concepto padre o concepto descompuesto del
50       * presupuesto.</i>
51       */
52      public String getParentConcept() {
53          return parentConcept;
54      }
55  
56      public void setParentConcept(String parentConcept) {
57          this.parentConcept = parentConcept;
58      }
59  
60      /**
61       * Spec definition:<br/>
62       * <i>CODIGO_HIJO: CODIGO del concepto hijo o concepto de la línea de
63       * descomposición.</i>
64       */
65      public String getConcept() {
66          return concept;
67      }
68  
69      public void setConcept(String concept) {
70          this.concept = concept;
71      }
72  
73      /**
74       * Spec definition:<br/>
75       * <i>POSICION: Posición del CONCEPTO_HIJO en la descomposición del
76       * CONCEPTO_PADRE, este dato permite identificar la medición cuando la
77       * descomposición del concepto padre incluye varios conceptos hijo con el
78       * mismo CODIGO, la numeración de las posiciones comenzará con el 1.
79       * 
80       * El campo POSICION deberá especificarse siempre en intercambio de
81       * presupuestos cuando éste sea completo y estructurado, e indicará el
82       * camino completo de la medición descrita en la estructura del archivo. Por
83       * ejemplo 3 \ 5 \ 2, indicará la medición correspondiente al capítulo 3 del
84       * archivo; subcapítulo 5 del capítulo 3; y partida 2 del subcapítulo 5. En
85       * mediciones no estructuradas este campo es opcional.</i>
86       */
87      public List<Integer> getPosition() {
88          return position;
89      }
90  
91      public void addPosition(Integer position) {
92          this.position.add(position);
93      }
94  
95      /**
96       * Spec definition:<br/>
97       * <i>MEDICION_TOTAL: Debe coincidir con el rendimiento del registro tipo
98       * '~D' correspondiente. Incorpora el sumatorio del producto de unidades,
99       * longitud, latitud y altura o el resultado de expresiones de cada línea,
100      * al leer este registro se recalculará este valor.</i>
101      */
102     public Float getTotal() {
103         return total;
104     }
105 
106     public void setTotal(Float total) {
107         this.total = total;
108     }
109 
110     /**
111      * Returns the lines of the measurement.
112      * 
113      * @return the lines of the measurement
114      */
115     public List<Line> getLines() {
116         return lines;
117     }
118 
119     /**
120      * Adds a new line to the measurement.
121      * 
122      * @return the new added line
123      */
124     public Line addLine() {
125         Line line = new Line();
126         this.lines.add(line);
127         return line;
128     }
129 
130     /**
131      * Spec definition:<br/>
132      * <i>ETIQUETA: Es opcional, surge de la necesidad de transmitir un
133      * identificador de los conceptos (capítulos, subcapítulos o partidas). Este
134      * identificador lo imprimen, diversos programas, en los listados de
135      * mediciones o presupuesto de una Obra (por ejemplo, ‘2.10’, ‘A-27b’,
136      * ‘001001’,…); siendo único para cada concepto (capítulo, subcapítulo o
137      * partida) y, en general, diferente de la codificación de la base de datos
138      * empleada para confeccionar el presupuesto (El ‘CODIGO_HIJO’ muchas veces
139      * no aparece en los listados mencionados). </i>
140      */
141     public String getLabel() {
142         return label;
143     }
144 
145     public void setLabel(String label) {
146         this.label = label;
147     }
148 
149     @Override
150     public String toString() {
151         return "Measurement {" + "Parent concept code: " + parentConcept
152                 + ", Concept code: " + concept + ", Position: " + position
153                 + ", Total: " + total + ", Label: " + label + ", Lines: "
154                 + lines + "}";
155     }
156 
157     /**
158      * A {@link Measurement} line.
159      * 
160      * @author DiSiD Team
161      */
162     public static class Line {
163 
164         Integer type;
165 
166         String comment;
167 
168         Float units;
169 
170         Float length;
171 
172         Float width;
173 
174         Float height;
175 
176         /**
177          * Spec definition:<br/>
178          * <i>/* TIPO: Indica el tipo de línea de medición de que se trate.
179          * Usualmente este subcampo estará vacío. Los tipos establecidos en esta
180          * VERSION son: '1': Subtotal parcial: En esta línea aparecerá el
181          * subtotal de las líneas anteriores desde el último subtotal hasta la
182          * línea inmediatamente anterior a ésta. '2': Subtotal acumulado: En
183          * esta línea aparecerá el subtotal de todas las líneas anteriores desde
184          * la primera hasta la línea inmediatamente anterior a ésta. '3':
185          * Expresión: Indicará que en el subcampo COMENTARIO aparecerá una
186          * expresión algebraica a evaluar. Se podrán utilizar los operadores
187          * '(', ')', '+', '-', '*', '/' y '^'; las variables 'a', 'b', 'c' y 'd'
188          * (que tendrán por valor las cantidades introducidas en los subcampos
189          * UNIDADES, LONGITUD, LATITUD y ALTURA respectivamente); y la constante
190          * 'p' para el valor Pi=3.1415926. Esta expresión será válida hasta la
191          * siguiente línea de medición en la que se defina otra expresión. Solo
192          * se evalúa la expresión y no se multiplica por las unidades. Las
193          * expresiones fórmulas utilizan los criterios definidos en el anexo 2.
194          * </i>
195          */
196         public Integer getType() {
197             return type;
198         }
199 
200         public void setType(Integer type) {
201             this.type = type;
202         }
203 
204         /**
205          * Spec definition:<br/>
206          * <i>COMENTARIO: Texto en la línea de medición. Podrá ser un comentario
207          * o una expresión algebraica.</i>
208          */
209         public String getComment() {
210             return comment;
211         }
212 
213         public void setComment(String comment) {
214             this.comment = comment;
215         }
216 
217         /**
218          * Spec definition:<br/>
219          * <i>UNIDADES, LONGITUD, LATITUD, ALTURA: Cuatro número reales con las
220          * mediciones. Si alguna magnitud no existe se dejará este campo
221          * vacío</i>
222          */
223         public Float getUnits() {
224             return units;
225         }
226 
227         public void setUnits(Float units) {
228             this.units = units;
229         }
230 
231         /**
232          * Spec definition:<br/>
233          * <i>UNIDADES, LONGITUD, LATITUD, ALTURA: Cuatro número reales con las
234          * mediciones. Si alguna magnitud no existe se dejará este campo
235          * vacío</i>
236          */
237         public Float getLength() {
238             return length;
239         }
240 
241         public void setLength(Float length) {
242             this.length = length;
243         }
244 
245         /**
246          * Spec definition:<br/>
247          * <i>UNIDADES, LONGITUD, LATITUD, ALTURA: Cuatro número reales con las
248          * mediciones. Si alguna magnitud no existe se dejará este campo
249          * vacío</i>
250          */
251         public Float getWidth() {
252             return width;
253         }
254 
255         public void setWidth(Float width) {
256             this.width = width;
257         }
258 
259         /**
260          * Spec definition:<br/>
261          * <i>UNIDADES, LONGITUD, LATITUD, ALTURA: Cuatro número reales con las
262          * mediciones. Si alguna magnitud no existe se dejará este campo
263          * vacío</i>
264          */
265         public Float getHeight() {
266             return height;
267         }
268 
269         public void setHeight(Float height) {
270             this.height = height;
271         }
272 
273         @Override
274         public String toString() {
275             return "Line {" + "Type: " + type + ", Comment: " + comment
276                     + ", Units: " + units + ", Length: " + length + ", Width: "
277                     + width + ", Height: " + height + "}";
278         }
279 
280     }
281 }
