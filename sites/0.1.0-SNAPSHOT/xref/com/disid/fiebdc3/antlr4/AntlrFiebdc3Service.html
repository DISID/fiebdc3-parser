19  package com.disid.fiebdc3.antlr4;
20  
21  import java.io.IOException;
22  import java.io.Reader;
23  
24  import org.antlr.v4.runtime.ANTLRInputStream;
25  import org.antlr.v4.runtime.CommonTokenStream;
26  import org.antlr.v4.runtime.LexerNoViableAltException;
27  import org.antlr.v4.runtime.tree.ParseTree;
28  import org.antlr.v4.runtime.tree.ParseTreeWalker;
29  
30  import com.disid.fiebdc3.Database;
31  import com.disid.fiebdc3.Fiebdc3Service;
32  import com.disid.fiebdc3.antlr4.Fiebdc3Parser;
33  
34  /**
35   * Implementation of a Fiebdc 3 file format parser service, based on ANTLR4.
36   * 
37   * @author DiSiD Team
38   */
39  public class AntlrFiebdc3Service implements Fiebdc3Service {
40  
41      @Override
42      public Database parse(Reader fiebdcData) throws IOException {
43          ANTLRInputStream input = new ANTLRInputStream(fiebdcData);
44          // create a lexer that feeds off of input CharStream
45          Fiebdc3Lexer lexer = new Fiebdc3Lexer(input) {
46              @Override
47              public void recover(LexerNoViableAltException e) {
48                  throw new RuntimeException(e); // Bail out
49              }
50          };
51          // create a buffer of tokens pulled from the lexer
52          CommonTokenStream tokens = new CommonTokenStream(lexer);
53          Fiebdc3Parser parser = new Fiebdc3Parser(tokens);
54          ParseTree tree = parser.database(); // begin parsing at database rule
55  
56          // Create a generic parse tree walker that can trigger callbacks
57          ParseTreeWalker walker = new ParseTreeWalker();
58          // Walk the tree created during the parse, trigger callbacks
59          Fiebdc3ListenerImpl listener = new Fiebdc3ListenerImpl();
60          walker.walk(listener, tree);
61  
62          return listener.getDatabase();
63      }
64  
65  }
