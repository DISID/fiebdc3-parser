19  package com.disid.fiebdc3.antlr4;
20  
21  import java.text.DateFormat;
22  import java.text.ParseException;
23  import java.text.SimpleDateFormat;
24  import java.util.Date;
25  
26  /**
27   * Utilities to parse values from string in the FIEBDC3 format.
28   * 
29   * @author DiSiD Team
30   */
31  public class ParseUtils {
32  
33      private static DateFormat fullDateFormat = new SimpleDateFormat("ddMMyyyy");
34      private static DateFormat mediumDateFormat = new SimpleDateFormat("ddMMyy");
35      private static DateFormat monthYearDateFormat = new SimpleDateFormat("MMyy");
36      private static DateFormat singleMonthYearDateFormat = new SimpleDateFormat(
37              "Myy");
38      private static DateFormat yearDateFormat = new SimpleDateFormat("yy");
39  
40      /**
41       * Parse dates with the format of FIEBDC3 dates:
42       * <p>
43       * <cite> DD representa el día con dos dígitos, MM el mes y AAAA el año. Si
44       * la fecha tiene 6 ó menos dígitos, el año se representará con dos dígitos
45       * (AA), interpretándose con el criterio “80/20”. Esto es, cualquier año que
46       * sea igual o superior a 80 corresponderá al siglo XX y cualquier año que
47       * sea menor de 80 corresponderá al siglo XXI. Si la fecha tiene menos de 5
48       * dígitos representa mes y año únicamente (MMAA), si tiene menos de tres,
49       * solo el año (AA). Si se identifica la fecha con un número impar de
50       * dígitos, se completará con el carácter cero por la izquierda. Para
51       * representar una fecha sin un día o mes específico, se utilizará un doble
52       * cero en cada caso. </cite>
53       * </p>
54       */
55      public static Date parseDate(String certDate) {
56          String dateToParse = certDate;
57          DateFormat dateFormat = fullDateFormat;
58          switch (certDate.length()) {
59          case 1:
60          case 2:
61              dateFormat = yearDateFormat;
62              break;
63          case 3:
64              dateFormat = singleMonthYearDateFormat;
65              break;
66          case 4:
67              dateToParse = replaceDoubleZero(dateToParse, 0);
68              dateFormat = monthYearDateFormat;
69              break;
70          case 5:
71          case 6:
72              dateToParse = replaceDoubleZero(dateToParse, 0);
73              dateToParse = replaceDoubleZero(dateToParse, 2);
74              dateFormat = mediumDateFormat;
75              break;
76          case 7:
77          case 8:
78          default:
79              dateToParse = replaceDoubleZero(dateToParse, 0);
80              dateToParse = replaceDoubleZero(dateToParse, 2);
81              dateFormat = fullDateFormat;
82          }
83  
84          try {
85              Date date = dateFormat.parse(dateToParse);
86              return date;
87          } catch (ParseException e) {
88              throw new RuntimeException(e);
89          }
90      }
91  
92      /**
93       * Replaces a "00" in a date String with a "01". This is because the FIEBDC3
94       * format allows to set the day or month value as "00" when it is ignored,
95       * but we are parsing a full date and so we need a valid one, and parsers
96       * don't admit "00" as day or month.
97       * 
98       * @param value
99       *            the string to perform the replacement on
100      * @param start
101      *            where to look for the "00" in te text
102      * @return a text with "00" value replaced if found
103      */
104     public static String replaceDoubleZero(String value, int start) {
105         if (value.regionMatches(start, "00", 0, 2)) {
106             String first = start > 0 ? value.substring(0, start) : "";
107             String last = value.substring(start + 2);
108             return first + "01" + last;
109         }
110         return value;
111     }
112 }
